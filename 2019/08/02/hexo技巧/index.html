     //创建草稿：
$ hexo new draft "new draft"

//草稿转文章（或手动移动_draft下相关文件至_post）：
$ hexo publish [layout] <filename>

//强行预览草稿，更改配置文件：
render_drafts: true

//启动草稿server：
$ hexo server --drafts
